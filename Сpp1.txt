#include <iostream> // Подключение библиотеки ввода-вывода для работы с консолью
using namespace std; // Использование пространства имен std, чтобы не писать std:: перед cout

int main() {
    // Массив с данными о вкладах, где каждый элемент - сумма вклада в долларах
    float a[8] = {100, 200, 440, 51, 96, 29, 47, 153};
    
    // Цикл для перебора всех элементов массива
    for (int i = 0; i < 8; i++) { // Начинаем с i = 0, идем до i < 8 (всего 8 элементов)
        if (a[i] >= 100) { // Проверяем, больше ли или равно текущее значение 100
            cout << a[i] << "$" << endl; // Если условие выполнено, выводим значение с символом "$"
        }
    }

    return 0; // Возврат 0, чтобы указать, что программа завершилась успешно
}
